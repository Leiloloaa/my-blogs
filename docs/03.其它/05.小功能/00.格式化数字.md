# 格式化数字

```javascript
 transform(value: any, ...args: any[]): any {
        if (!value) {
            return value;
        } else {
            // 向上取整
            value = Math.floor(value * 100) / 100 + '';
            const xsd = value.split('.');
            // 等于
            if (xsd.length === 1) {
                value = value + '.00';
                if (xsd[0].length > 3) {
                    xsd[0] = xsd[0].replace(/\B(?=(?:\d{3})+\b)/g, ',');
                    value = xsd[0] + '.00';
                }
                return value;
            }
            // 大于一位
            if (xsd.length > 1) {
                if (xsd[0].length > 3) {
                    xsd[0] = xsd[0].replace(/\B(?=(?:\d{3})+\b)/g, ',');
                    value = xsd[0] + '.' + xsd[1];
                }
                // 小于两位
                if (xsd[1].length < 2) {
                    value = value + '0';
                }
                return value;
            }

        }

    }
```
